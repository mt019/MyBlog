12async function copyCodeToClipboard(button, highlightDiv) {
13  const codeToCopy = Array.from(highlightDiv.querySelectorAll(":last-child > .chroma > code > .line >.cl")) 
14  //嗷嗷解決了！！！終於不會copy到line number！
15  .map(codeElement => codeElement.innerText)
16  .join(" ");
17  try {
18    result = await navigator.permissions.query({ name: "clipboard-write" });
19    if (result.state == "granted" || result.state == "prompt") {
20      await navigator.clipboard.writeText(codeToCopy);
21    } else {
22      copyCodeBlockExecCommand(codeToCopy, highlightDiv);
23    }
24  } catch (_) {
25    copyCodeBlockExecCommand(codeToCopy, highlightDiv);
26  } finally {
27    codeWasCopied(button);
28  }
29}

31function copyCodeBlockExecCommand(codeToCopy, highlightDiv) {
32  const textArea = document.createElement("textArea");
33  textArea.contentEditable = "true";
34  textArea.readOnly = "false";
35  textArea.className = "copyable-text-area";
36  textArea.value = codeToCopy;
37  highlightDiv.insertBefore(textArea, highlightDiv.firstChild);
38  const range = document.createRange();
39  range.selectNodeContents(textArea);
40  const sel = window.getSelection();
41  sel.removeAllRanges();
42  sel.addRange(range);
43  textArea.setSelectionRange(0, textArea.value.length);
44  // document.execCommand("copy");
45  highlightDiv.removeChild(textArea);
46}

48function codeWasCopied(button) {

54}

56function addCopyButtonToDom(button, highlightDiv) {

62}

 1// When the user scrolls down 20px from the top of the document, show the button
 2window.onscroll = function() {scrollTopButton()};
 3
 4function scrollTopButton() {
 5  if (document.body.scrollTop > 20 || document.documentElement.scrollTop > 20) {
 6    document.getElementById("goTopBtn").style.display = "block";
 7  } else {
 8    document.getElementById("goTopBtn").style.display = "none";
 9  }
10}
11
12// When the user clicks on the button, scroll to the top of the document
13function scrollTop() {
14  document.body.scrollTop = 0; // For Safari
15  document.documentElement.scrollTop = 0; // For Chrome, Firefox, IE and Opera
16}
17
18//Smooth scroll to the top of the document
19function smoothScrollTop(){
20    var timer  = null;
21    cancelAnimationFrame(timer);
22    timer = requestAnimationFrame(function fn(){
23        var oTop = document.body.scrollTop || document.documentElement.scrollTop;
24        if(oTop > 0){
25            document.body.scrollTop = document.documentElement.scrollTop = oTop - 50;
26            timer = requestAnimationFrame(fn);
27        }else{
28            cancelAnimationFrame(timer);
29        }
30    });
31}
